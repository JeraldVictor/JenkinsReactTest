import { takeLatest, takeEvery } from 'redux-saga/effects';
 
import * as journeyUserActions from '../../modules/journey-user/users/actions/user_action_types';
import * as journeyGroupActions from '../../modules/journey-user/groups/actions/groups_action_types';
 
import * as journeyUserWorkerSaga from './journey_user_worker_saga';
 
export function* journeyUserWatcherSaga() {
  yield takeLatest(
    journeyUserActions.REQUEST_JOURNEY_USER_LIST,
    journeyUserWorkerSaga.getJourneyUserList,
  );
  yield takeLatest(
    journeyGroupActions.REQUEST_GROUP_LIST,
    journeyUserWorkerSaga.getGroupList,
  );
  yield takeLatest(
    journeyUserActions.REQUEST_USER_DETAILS,
    journeyUserWorkerSaga.getUserDetails,
  );
  yield takeEvery(
    journeyUserActions.UPDATE_USER_PROFILE_DETAILS,
    journeyUserWorkerSaga.updateUserProfileDetails,
  );
  yield takeLatest(
    journeyUserActions.REQUEST_USER_JOURNEY_LIST,
    journeyUserWorkerSaga.getUserJourneyList,
  );
  yield takeLatest(
    journeyUserActions.REQUEST_USER_SESSION_LIST,
    journeyUserWorkerSaga.getUserSessionList,
  );
  yield takeLatest(
    journeyUserActions.REQUEST_USER_GROUP_DATA,
    journeyUserWorkerSaga.getUserGroupList,
  );
  yield takeLatest(
    journeyUserActions.SEND_USER_GROUP_DATA,
    journeyUserWorkerSaga.setUserGroupList,
  );
  yield takeLatest(
    journeyUserActions.GET_AVAILABLE_ROLES,
    journeyUserWorkerSaga.getAvailableRoles,
  );
  yield takeLatest(
    journeyUserActions.IMPORT_USER_LIST,
    journeyUserWorkerSaga.importUsersList,
  );
  yield takeLatest(
    journeyUserActions.CANCEL_IMPORT_USER_LIST,
    journeyUserWorkerSaga.importUsersList,
  );
  yield takeEvery(
    journeyUserActions.CANCEL_ASSIGNED_JOURNEY_LIST,
    journeyUserWorkerSaga.cancelJourneyAssigned,
  );
  yield takeLatest(
    journeyUserActions.UPDATE_USER_GROUP_ASSOCIATION,
    journeyUserWorkerSaga.updateUserGroupAssociations,
  );
  yield takeLatest(
    journeyUserActions.CHECK_FOR_UNIQUE_EMAIL,
    journeyUserWorkerSaga.checkForEmailUniqueness,
  );
  yield takeLatest(
    journeyUserActions.CHECK_FOR_UNIQUE_USERNAME,
    journeyUserWorkerSaga.checkForUsernameUniqueness,
  );
  yield takeLatest(
    journeyUserActions.ADD_NEW_USER,
    journeyUserWorkerSaga.saveNewUserDetails,
  );
  yield takeLatest(
    journeyUserActions.IMPORT_GROUP_USERS,
    journeyUserWorkerSaga.importGroupUsers,
  );
  yield takeEvery(
    journeyUserActions.REMOVE_USER_GROUP_ASSOCIATION,
    journeyUserWorkerSaga.removeUserGroupAssociation,
  );
  yield takeLatest(
    journeyUserActions.CHECK_FOR_UNIQUE_EMPLOYEE_ID,
    journeyUserWorkerSaga.checkForEmployeeIdUniqueness,
  );
  yield takeLatest(
    journeyUserActions.GET_JOURNEY_ACTIVATION_STATUS,
    journeyUserWorkerSaga.getJourneyStatusForUserProfile,
  );
  yield takeLatest(
      journeyUserActions.EXPORT_USERS,
      journeyUserWorkerSaga.exportUsers,
  );
  yield takeLatest(
    journeyUserActions.GET_USER_JOURNEY_LATEST_VERSIONS,
    journeyUserWorkerSaga.getUserJourneyLatestVersions,
  );
}
