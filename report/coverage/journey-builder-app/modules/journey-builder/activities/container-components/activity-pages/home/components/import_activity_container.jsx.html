import React, { useEffect, useState } from 'react';
import ActionHeader from 'terra-action-header';
import Grid from 'terra-grid';
import SearchField from 'terra-search-field';
import ContentContainer from 'terra-content-container';
import { withDisclosureManager } from 'terra-disclosure-manager';
import connect from 'react-redux/es/connect/connect';
import PropTypes from 'prop-types';
import ReactDOM from 'react-dom';
import { bindActionCreators } from 'redux';
import IconSpinner from 'terra-icon/lib/icon/IconSpinner';
 
import ActivityList from './activity_list';
import { searchActivityImportPlaceHolder, importActivityLabel } from '../../../util/Constants';
import { getOrgPublishedActivities, searchPublishedActivity } from '../../../../actions/activity_actions';
import SourceLibrary from '../../../../../../../reusable_components/source_library';
import DuplicateActivityContainer from './duplicate_activity_container';
import Loading from "../../../../../../../utils/loading";
 
const ImportActivityContainer = (props) => {
  const { disclosureManager } = props;
  const { orgSourceLibraries } = props.orgListDetailsReducer;
  const { isPublishedActivitiesLoading } = props.activitiesReducer;
  const [displayNoActivityFound, setDisplayNoActivityFound] = useState(false);
  const [searchString, setSearchString] = useState('');
  let activitySearchModalRef = null;
 
  const defaultSourceLibrary = orgSourceLibraries.length > 0 &&
        ((props.persistedDetails.orgId === orgSourceLibraries[0].org_id) && orgSourceLibraries.length > 1 ?
            orgSourceLibraries[1].org_id : orgSourceLibraries[0].org_id);
  const [currentSelectedOrg, setCurrentSelectedOrg] = useState(defaultSourceLibrary);
 
  useEffect(() => {
    props.getOrgPublishedActivities(defaultSourceLibrary);
  }, [],
  );
 
  const fetchActivityList = (e) => {
    const searchInput = e;
    if (searchInput !== undefined && (searchInput.length > 2 || searchInput.length === 0)) {
      props.searchPublishedActivity(currentSelectedOrg, searchString);
    }
  };
 
  const validateActivitySearchInput = (searchText) => {
    setSearchString(searchText);
    const searchMagnifyIcon = ReactDOM.findDOMNode(activitySearchModalRef).children[1];
    if (searchText === undefined || searchText.length < 3) {
      searchMagnifyIcon.setAttribute('disabled', true);
    } else {
      searchMagnifyIcon.removeAttribute('disabled');
    }
    if (!searchMagnifyIcon.getAttribute('disabled')) {
      setDisplayNoActivityFound(true);
    }
  };
 
  const handleInvalidActivitySearch = () => {
    props.getOrgPublishedActivities(currentSelectedOrg);
  };
 
  const loadOrgActivities = (orgId) => {
    setCurrentSelectedOrg(orgId);
    props.getOrgPublishedActivities(orgId);
    setDisplayNoActivityFound(false);
  };
 
  const closeLoadModal = () => {
    disclosureManager.closeDisclosure();
  };
 
  const importActivity = () => {
    const { importStatus, isActivityImportInProgress, duplicateActivities, hasRemovedDuplicateActivities } = props.activitiesReducer;
 
    if (isActivityImportInProgress === true) {
      return (
        <IconSpinner className="right-button automation_loading_icon" />
      );
    } else if (duplicateActivities.length > 0 || hasRemovedDuplicateActivities) {
      return (<DuplicateActivityContainer handleCloseModal={disclosureManager.closeDisclosure} />);
    } else if (importStatus === 1 || importStatus === -1) {
      closeLoadModal();
    } else {
      return (<ContentContainer
        fill
        header={(
          <div className="import-activity-header">
            <ActionHeader
              title={importActivityLabel}
              onClose={disclosureManager.closeDisclosure}
            />
          </div>
              )}
      >
        <div className="load-activity-modal-container">
          <Grid>
            <Grid.Row>
              <Grid.Column tiny={6} small={6} medium={6} large={6} huge={6}>
                <SearchField
                  className="search-field-modal"
                  id="activitySearchModal"
                  ref={el => activitySearchModalRef = el}
                  minimumSearchTextLength={3}
                  placeholder={searchActivityImportPlaceHolder}
                  disableAutoSearch
                  onSearch={fetchActivityList}
                  onInvalidSearch={handleInvalidActivitySearch}
                  onChange={e => validateActivitySearchInput(e.target.value)}
                />
              </Grid.Column>
              <Grid.Column tiny={6} small={6} medium={6} large={6} huge={6}>
                <SourceLibrary
                  orgSourceLibraries={orgSourceLibraries}
                  defaultSourceLibrary={defaultSourceLibrary}
                  loadSourceOrgData={loadOrgActivities}
                />
              </Grid.Column>
            </Grid.Row>
          </Grid>
          {isPublishedActivitiesLoading ?
            <div><Loading padding="13.5" /></div> :
            <ActivityList
              handleCloseModal={disclosureManager.closeDisclosure}
              displayNoActivityFound={displayNoActivityFound}
            />}
        </div>
      </ContentContainer>);
    }
  };
 
  return (
    <React.Fragment>
      {importActivity()}
    </React.Fragment>
  );
};
 
const mapDispatchToProps = dispatch => ({
  getOrgPublishedActivities: bindActionCreators(getOrgPublishedActivities, dispatch),
  searchPublishedActivity: bindActionCreators(searchPublishedActivity, dispatch),
 
});
 
const mapStateToProps = state => ({
  persistedDetails: state.persistedDetails,
  orgListDetailsReducer: state.orgListDetailsReducer,
  activitiesReducer: state.activitiesReducer,
});
 
ImportActivityContainer.propTypes = {
  disclosureManager: PropTypes.func,
  orgListDetailsReducer: PropTypes.func,
  persistedDetails: PropTypes.func,
};
 
export default withDisclosureManager(connect(mapStateToProps, mapDispatchToProps)(ImportActivityContainer));
