import { richTextCharacterLimit } from './Constants';
import cloneDeep from 'lodash/cloneDeep';
 
export const handleCharacterLimit = (delta, textEditorRef) => {
  if (textEditorRef !== null && textEditorRef !== undefined) {
    const editorText = textEditorRef.getEditor().root.innerHTML;
 
    const charactersOver = (editorText.length) - richTextCharacterLimit;
 
    const { ops } = delta;
    let updatedOps;
 
    if (charactersOver > 0) {
      if (ops[0].insert) {
        const charactersToRetain = ops[0].insert.length - charactersOver;
        if (charactersToRetain === 0) {
          updatedOps = [{ delete: charactersOver }];
        } else {
          updatedOps = [{ retain: charactersToRetain }, { delete: charactersOver }];
        }
      } else {
        const charactersToRetain = ops[1].insert.length - charactersOver;
        updatedOps = [{ retain: ops[0].retain + charactersToRetain }, { delete: charactersOver }];
      }
      textEditorRef.getEditor().updateContents({ ops: updatedOps });
    }
  }
};

/*
    Return ungrouped activities and formed activity groups(if groups exist)
 */
export const buildStageActivities = (stageId, stage_activity_details, addStageActivityGroups) => {
  let stageActivities = cloneDeep(stage_activity_details);
  const groups = {};
  let isActivityGroupPresent = false;
 
  // Form activity groups if any exists
  stageActivities.forEach((activity, index) => {
    if (activity.activity_group_name) {
      const groupName = activity.activity_group_name;
      const id = new Date().valueOf() + index;
      groups[`${groupName}`] = groups[`${groupName}`] ? groups[`${groupName}`] : {
        id,
        activity_name: groupName,
        activity_type: 'Activity Group',
      };
      // subRows contain the activities within the group and is required key to display subRows for React table
      groups[`${groupName}`].subRows = groups[`${groupName}`].subRows ? groups[`${groupName}`].subRows
        : [];
      groups[`${groupName}`].subRows.push(activity);
 
      if (groups[`${groupName}`].index === undefined) {
        groups[`${groupName}`].index = index;
      } else {
        groups[`${groupName}`].index = groups[`${groupName}`].index > index ? index
          : groups[`${groupName}`].index;
      }
 
      isActivityGroupPresent = true;
    }
  });
 
  if (isActivityGroupPresent) {
    // Replace grouped activity with the group
    for (const group in groups) {
      stageActivities.splice(groups[group].index, 1);
      stageActivities.splice(groups[group].index, 0, groups[group]);
    }
 
    // Remove grouped activities from list  as they are already added as part of group above
    stageActivities = stageActivities.filter(activity => !activity.activity_group_name);
 
    // Update existing activity groups in stage to be able to check unique activity group name
    addStageActivityGroups(stageId, groups);
  }
 
  return stageActivities;
};
