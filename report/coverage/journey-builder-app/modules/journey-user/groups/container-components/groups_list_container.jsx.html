import React from 'react';
import Grid from 'terra-grid';
import { connect } from 'react-redux';
import { bindActionCreators } from 'redux';
import { NavLink } from 'react-router-dom';
import IconAdd from 'terra-icon/lib/icon/IconAdd';
import SearchField from 'terra-search-field';
import Alert from 'terra-alert';
import ReactTooltip from 'react-tooltip';
import groupActions from '../actions/group_actions';
import { successMessage } from '../../../journey-builder/journeys/container-components/Utils/Constants';
import { TECHNICAL_ERROR_MESSAGE, GROUP_DELETED_SUCCESS_MESSAGE, GROUP_ASSIGNMENT_IN_PROGRESS_MESSAGE } from '../../../../utils/user_module_constants';
import GroupList from '../presentaional-components/group_list';
import Loading from '../../../../utils/loading';
import { supportAdminTooltipInfo } from "../../../../utils/Constants"
import ExportErrorList from '../../../../reusable_components/export_data_to_xlsx';
import { FAILED_GROUP_COLUMN_NAMES } from '../../../../utils/group_module_constants';
import SearchHelper from '../helper/search_helper';
import isEmpty from 'lodash/isEmpty';
 
export class GroupsContainer extends React.Component {
  constructor(props) {
    super(props);
    this.state = {
      searchString: '',
      searchStringLength: null,
      showDeleteWarningMsg: true,
      showReenabledWarningMsg: true,
    };
  }
  componentDidMount() {
    // set 0 as stepper position whenever there is visit to group list or revisit to other group
    this.props.groupActions.setStepperPosition(0);
    this.props.groupActions.setJourneyAssignmentInProgress(false);
 
    if (isEmpty(this.props.journeyGroupReducer.retain_group_user_search) || !this.props.journeyGroupReducer.isSearchToBeRetained) {
      this.getGroupList();
    }
    else {
      this.setState({ searchString: this.props.journeyGroupReducer.retain_group_user_search.search_string });
      this.props.groupActions.getGroupList(this.props.journeyGroupReducer.retain_group_user_search);
    }
 
    const searchIcon = document
      .getElementById('groupSearch')
      .querySelector('button[title="Submit Search"]');
    searchIcon.setAttribute('disabled', true);
  }
 
  componentWillUnmount() {
    this.handleDismiss();
    this.props.groupActions.resetGroupListPageAttributes();
  }
 
  handleDismiss = () => {
    this.props.groupActions.dismissGroupsAlert();
  };
 
  onChangeSearchString = (e) => {
    const searchEntry = e.target.value;
    this.setState({ searchString: searchEntry });
  
    const searchMagnifyIcon = document
      .getElementById('groupSearch')
      .querySelector('button[title="Submit Search"]');
    if (searchEntry === ''|| searchEntry === undefined ) {
      searchMagnifyIcon.setAttribute('disabled', true);
      this.props.groupActions.resetGroupListPageAttributes();
      this.props.groupActions.getGroupList(SearchHelper.customSearch(this.props.persistedDetails.orgId, ''));
    } else if (searchEntry.length < 1) {
      searchMagnifyIcon.setAttribute('disabled', true);
    } else {
      searchMagnifyIcon.removeAttribute('disabled');
    }
  };
 
  getGroupList = (e) => {
    const searchInput = e;
    const orgId = this.props.persistedDetails.orgId;
    this.setState({
      searchStringLength: this.state.searchString.length,
    });
    if (searchInput != undefined) {
      this.props.groupActions.resetGroupListPageAttributes();
      this.props.groupActions.getGroupList(SearchHelper.customSearch(this.props.persistedDetails.orgId, this.state.searchString));
    }
    else if (searchInput == undefined) {
      this.props.groupActions.resetGroupListPageAttributes();
      this.props.groupActions.getGroupList(SearchHelper.customSearch(this.props.persistedDetails.orgId, this.state.searchString));
    }
  };
 
  render() {
    const { journeyGroupReducer } = this.props;
    return (
      <React.Fragment>
        <Grid>
          <Grid.Row>
            <Grid.Column
              tiny={12}
              small={12}
              medium={12}
              large={12}
              huge={12}
              className="instruction-text"
            >
              {this.props.persistedDetails.privileges.includes('ADD_GROUP') && !this.props.persistedDetails.privileges.includes('VIEW_FOR_SUPPORT_ADMIN') ?
                <a href="https://wiki.cerner.com/x/PZPOdw" target="_blank">
                  Add Group Instructions
                </a> : <a href="#" style={{ "opacity": "0.4" }}>
                  Add Group Instructions
                </a>
              }
            </Grid.Column>
          </Grid.Row>
          <Grid.Row>
            <Grid.Column
              tiny={4}
              small={4}
              medium={4}
              large={4}
              huge={4}
              className="journey-user-action-buttons"
            >
              <ReactTooltip />
              {this.props.persistedDetails.privileges.includes('ADD_GROUP') && <span data-tip={supportAdminTooltipInfo} data-tip-disable={!this.props.persistedDetails.privileges.includes('VIEW_FOR_SUPPORT_ADMIN')} className={this.props.persistedDetails.privileges.includes('VIEW_FOR_SUPPORT_ADMIN') && "disabled-import"}>
                <NavLink
                  to={`${this.props.match.url}/create_group`} onClick={e => this.props.persistedDetails.privileges.includes('VIEW_FOR_SUPPORT_ADMIN') &&
                    e.preventDefault()}
                >
                  <IconAdd /> Add Group{' '}
                </NavLink> </span>
              }
            </Grid.Column>
            <Grid.Column tiny={8} small={8} medium={8} large={8} huge={8}>
              {this.props.persistedDetails.privileges.includes('SEARCH_VIEW_GROUP') &&
                <SearchField
                  className="search-field-group"
                  disableAutoSearch
                  minimumSearchTextLength={1}
                  id="groupSearch"
                  placeholder="Search by group name"
                  value={this.state.searchString}
                  onSearch={this.getGroupList}
                  onChange={this.onChangeSearchString}
                />
              }
            </Grid.Column>
          </Grid.Row>
          <Grid.Row>
            <Grid.Column tiny={12} small={12} medium={12} large={12} huge={12}>
              {/* Group List Fetch Error */}
              {this.props.journeyGroupReducer.groupListFetchError &&
                <Alert type="error">{TECHNICAL_ERROR_MESSAGE}</Alert>}
 
              {this.props.journeyGroupReducer.groupDeleteAttributes.showGroupDeleteSuccessMsg &&
                // Alert on successful Group Delete
                <Alert
                  type="success"
                  title="Success!"
                  className="group-alert"
                  onDismiss={this.handleDismiss}
                >
                  {GROUP_DELETED_SUCCESS_MESSAGE}
                </Alert>}
 
              {/* Alert for Group Assignment in Progress */}
              {(this.state.showDeleteWarningMsg && this.props.journeyGroupReducer.groupDeleteAttributes.showGroupAssignmentMsg) &&
                <Alert type="warning" id="warning-msg" className="group-alert" onDismiss={() => { this.setState({ showDeleteWarningMsg: false }); }}>{GROUP_ASSIGNMENT_IN_PROGRESS_MESSAGE}</Alert>}
 
              {(this.props.journeyGroupReducer.assignJourneyStatus != null || this.props.journeyGroupReducer.showSaveAssignmentAlert) && (
                <Alert
                  type="success"
                  title="Success!"
                  className="group-alert"
                  onDismiss={this.handleDismiss}
                >
                  {successMessage}
                </Alert>
              )}
              {(this.state.showReenabledWarningMsg && this.props.journeyGroupReducer.failedGroupList.length > 0) &&
                <Alert type="warning" id="warning-msg" className="group-alert" onDismiss={() => { this.setState({ showReenabledWarningMsg: false }) }}>
                  <span>An error occurred and one or more groups were not deleted. Click&nbsp;
                    <NavLink to="#">
                      <ExportErrorList button={<span>here </span>}
                        columnNames={FAILED_GROUP_COLUMN_NAMES}
                        rowData={this.props.journeyGroupReducer.failedGroupList}
                        filename="failed-group-list"
                        excelSheetName="FailedGroupList"
                      />
                    </NavLink>to view the list.
                  </span></Alert>}
            </Grid.Column>
          </Grid.Row>
          <Grid.Row>
            <Grid.Column tiny={12} small={12} medium={12} large={12} huge={12}>
              {this.props.persistedDetails.privileges.includes('SEARCH_VIEW_GROUP') &&
                <React.Fragment>
                  {journeyGroupReducer.isGroupListLoading ? (
                    <div className="group_details_loader"><Loading padding="13.5" /></div>
                  ) : (
                    <GroupList
                      groupList={this.props}
                      searchStringLength={this.state.searchStringLength}
                    />
                  )}
                </React.Fragment>
              }
            </Grid.Column>
          </Grid.Row>
        </Grid>
      </React.Fragment>
    );
  }
}
 
const mapDispatchToProps = dispatch => ({
  groupActions: bindActionCreators(groupActions, dispatch),
});
 
const mapStateToProps = state => ({
  journeyGroupReducer: state.journeyGroupReducer,
  persistedDetails: state.persistedDetails,
});
 
export default connect(mapStateToProps, mapDispatchToProps)(GroupsContainer);
