import React, { useEffect, useState } from "react";
import { useSelector, useDispatch } from "react-redux";
import ActionFooter from "terra-action-footer";
import ActionHeader from "terra-action-header";
import Button from "terra-button/lib/Button";
import ContentContainer from "terra-content-container";
import { withDisclosureManager } from "terra-disclosure-manager";
import Loading from "../utils/loading";
import commonActions from "../common-actions/common_actions";
import moment from "moment";
import {
  NOTES_DATE_TIME_FORMAT,
  NOTES_TITLE,
  READ_ONLY,
  CLOSE,
  SHOW_NOTES_FOOTER_START_MSG,
  SHOW_NOTES_FOOTER_END_MSG,
  CANCEL,
  ADD,
  NOTES_SERVICE_ERROR_MSG,
  PREVIEW_USER,
} from "../utils/notes_constants";
import { NavLink } from "react-router-dom";
import Alert from "terra-alert";
import {
  formats,
  modules,
  notesDescriptionPlaceHolder,
} from "../modules/journey-builder/activities/container-components/util/Constants";
import ReactQuill from "react-quill";
import { handleCharacterLimitEvent } from "../modules/journey-builder/events/container-components/utils/event_util";
import {
  charactersRemainingMsg,
  richTextEventsLimit,
} from "../modules/journey-builder/events/container-components/utils/event_constants";
import ScrollToBottom, { useScrollToBottom, useSticky } from 'react-scroll-to-bottom';
import {stripHtmlFromString} from "../modules/journey-builder/activities/container-components/util/activity_util";
 
const ShowNotesModal = (props) => {
  const {
    disclosureManager,
    entityId,
    entityName,
    // activityType is used only for Activity module for accessing detail page, rest modules do not use.
    activityType,
    addNotes,
    notesBody,
    updateNotesDescription,
  } = props;
 
  const fullName = props.fullName && props.fullName;
  const scrollToBottom = useScrollToBottom();
  const [sticky] = useSticky();
 
  const [notesRef, setNotesRef] = useState("");
  const [notesFieldRef, setNotesFieldRef] = useState(null);
  const [note, setNote] = useState(notesBody);
  const [addButtonIsDisabled, SetAddButtonIsDisabled] = useState(true);
 
  const common = useSelector((state) => state.common);
  const dispatch = useDispatch();
  const persistedDetails = useSelector((state) => state.persistedDetails);
  const noteLength = notesRef && notesRef.editor && richTextEventsLimit + 1 - notesRef.editor.getLength();
 
  const notes_privileges = !(persistedDetails.privileges.includes('ADD_EVENT') || persistedDetails.privileges.includes('UPDATE_EVENT') || persistedDetails.privileges.includes('ADD_ACTIVITY') || persistedDetails.privileges.includes('UPDATE_ACTIVITY') || persistedDetails.privileges.includes('ADD_JOURNEY'))  || persistedDetails.privileges.includes('VIEW_FOR_SUPPORT_ADMIN');
 
  // below footer_link will be used conditionally for other modules to make it dynamic
  const footer_link =
    entityName === "Activity"
      ? `/journey_builder/activities/edit/${activityType}/${entityId}`
      : entityName === "Event"
      ? `/journey_builder/events/${entityId}`
      : entityName === "Journey"
      ? `/journey_builder/journeys/${entityId}`
      : '';
 
  const handleSaveNotes = () => {
    const save_notes_details = {
      entity_id: entityId,
      user_id: persistedDetails.userId,
      body: note,
    };
 
    dispatch(commonActions.saveNotes(save_notes_details, entityName));
 
    setToDefaultState()
  };
 
  const attachQuillRefs = (el, richTextField) => {
    richTextField === "notes" ? el !== notesRef && setNotesRef(el) : "";
  };
 
  const handleAddNotes = () => {
    if (notesRef && notesRef.editor) {
      SetAddButtonIsDisabled(
        !(notesRef.editor.getLength() > 1 && notesRef.editor.getLength() <= richTextEventsLimit + 1)
      );
    }
  };
 
  const setToDefaultState = () => {
    setNote(null);
    SetAddButtonIsDisabled(true);
  }
 
  useEffect(() => {
    if (common.isNotesSaved && (note != notesBody)) {
      setToDefaultState();
    }
  }, [common]);
 
 
  useEffect(() => {
    if (entityId) {
      dispatch(commonActions.showNotesLoader());
      dispatch(commonActions.fetchNotes(entityId, entityName));
    }
    SetAddButtonIsDisabled(note ? false : true);
  }, []);
 
  return (
    <ContentContainer
      header={
        <ActionHeader
          id="notes_header_footer"
          level={2}
          title={entityName + NOTES_TITLE + (addNotes ? '' : READ_ONLY)+(entityName === 'Competency' ? ` for ${fullName}` : '')}
          onClose={disclosureManager.closeDisclosure}
        />
      }
      footer={
        <div>
          {addNotes === true && !common.isNotesLoading && (
            <div className="notes_quill">
              <ReactQuill
                className="add_notes_editor"
                formats={formats}
                modules={modules}
                placeholder={notesDescriptionPlaceHolder}
                ref={(el) => {
                  setNotesFieldRef(el);
                  el && el.editor && attachQuillRefs(el, "notes");
                }}
                value={note}
                onChange={(content, delta) => {
                  setNote(content);
                  updateNotesDescription(content);
                  handleCharacterLimitEvent(delta, notesFieldRef);
                  handleAddNotes();
                }}
              />
 
              <span className="notes_character_counter">
                {charactersRemainingMsg}{" "}
                <span className={noteLength <= 0 ? "error" : "character-count"}>
                  {noteLength < 0 ? "0" : noteLength}
                </span>
              </span>
            </div>
          )}
 
          {!common.isNotesLoading &&
          <ActionFooter
              id="notes_header_footer"
              start={
                <>
                  {/* addNotes is true only when add notes through detail page */}
                  {addNotes !== true ? (
                      <div id="activityDetailLink">
                        {SHOW_NOTES_FOOTER_START_MSG}
                        <NavLink activeClassName="active" to={footer_link}>
                      <span onClick={disclosureManager.closeDisclosure}>
                        {entityName}
                      </span>
                        </NavLink>
                        {SHOW_NOTES_FOOTER_END_MSG}
                      </div>
                  ) : (
                      ""
                  )}
                </>
              }
              end={
                <>
                  <Button
                      text={addNotes === true ? CANCEL : CLOSE}
                      className="right_button_spacing"
                      onClick={disclosureManager.closeDisclosure}
                  />
 
                  {addNotes === true ? (
                      <Button
                          text={ADD}
                          className="right_button_spacing"
                          variant="emphasis"
                          isDisabled={addButtonIsDisabled || (stripHtmlFromString(note).length <= 0) || notes_privileges}
                          onClick={() => handleSaveNotes()}
                      />
                  ) : (
                      ""
                  )}
                </>
              }
          />
          }
        </div>
      }
    >
      {common.isNoteServiceDown && (
        <div className={`activity_modal_body ${addNotes === true ? "add_modal_height" : "show_modal_height"}`}>
        <Alert type="error" className="service_down_notes">
            {NOTES_SERVICE_ERROR_MSG}
        </Alert>
        </div>
      )}
      {common.isNotesLoading ? (
        <div className={`activity_modal_body ${addNotes === true ? "add_modal_height" : "show_modal_height"}`}>
          <Loading padding="1.5" border="none" />
        </div>
      ) : (
 
        <ScrollToBottom initialScrollBehavior="smooth" mode="bottom" followButtonClassName="arrow down">
          <div className={`activity_modal_body ${addNotes === true ? "add_modal_height" : "show_modal_height"}`}>
            {common.notes?.map((data) => {
                return (
                  <div key={data.id} className="activity_note_content">
                    <p className="activity_note_date">
                      {moment(new Date(data.created_at))
                        .tz(persistedDetails.timeZoneRegion)
                        .format(NOTES_DATE_TIME_FORMAT)}{" "}
                      by {data.full_name ? data.full_name : PREVIEW_USER}
                    </p>
                    <div
                      className={"activity_note_body"}
                      dangerouslySetInnerHTML={{ __html: data.body }}
                    />
                  </div>
                );
              })}
              {!sticky && <button onClick={scrollToBottom}>Click me to scroll to bottom</button>}
          </div>
        </ScrollToBottom>
      )}
    </ContentContainer>
  );
};
 
export default withDisclosureManager(ShowNotesModal);
