export const handleIncompleteDuration = (minutes, seconds) => {
  // Converting explicitly to integer for interpreting empty value and null correctly as not a number
  let min = parseInt(minutes, 10);
  let sec = parseInt(seconds, 10);
 
  // Duration is not provided
  if (isNaN(min) && isNaN(sec)) {
    return null;
  }
 
  // Handle partial (only mins or secs provided) duration input
  if (!isNaN(min) && isNaN(sec)) {
    sec = 0;
  } else if (isNaN(min) && !isNaN(sec)) {
    min = 0;
  }
 
  // Avoid duration exceeding the max allowed value of 60 minutes
  if (min === 60 && sec === 60) {
    sec = 0;
  }
 
  return (min * 60) + sec;
};

export const handleCharacterLimit = (delta, textEditorRef) => {
  if (textEditorRef !== null && textEditorRef !== undefined) {
    const editorText = textEditorRef.getEditor().getText();
    const charactersOver = (editorText.length - 1) - richTextCharacterLimit;
 
    const { ops } = delta;
    let updatedOps;
 
    if (charactersOver > 0) {
      if (ops[0].insert) {
        const charactersToRetain = ops[0].insert.length - charactersOver;
        if (charactersToRetain === 0) {
          updatedOps = [{ delete: charactersOver }];
        } else {
          updatedOps = [{ retain: charactersToRetain }, { delete: charactersOver }];
        }
      } else {
        const charactersToRetain = ops[1].insert.length - charactersOver;
        updatedOps = [{ retain: ops[0].retain + charactersToRetain }, { delete: charactersOver }];
      }
      textEditorRef.getEditor().updateContents({ ops: updatedOps });
    }
  }
};
