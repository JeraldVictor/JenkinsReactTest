'Import failed!' : journeyUserReducer.importStatus === 'INVALID_IMP_TEMPLATE' ? userModuleConstants.invalidImpTemplateTitle : ''}
                  className="alert-message"
                  onDismiss={journeyUserReducer.importStatus === 'ERROR' || journeyUserReducer.importStatus === 'NETWORK_ERROR' || journeyUserReducer.importStatus === 'INVALID_IMP_TEMPLATE' ? this.handleAlertMessage : journeyUserReducer.importStatus === 'SUCCESS' ? this.handleAlertMessageOnSuccess : null}
                >
                  {journeyUserReducer.importStatus === 'READY_TO_IMPORT' ? userModuleConstants.READY_TO_IMPORT :
                    journeyUserReducer.importStatus === 'NETWORK_ERROR' ? userModuleConstants.TECHNICAL_ERROR_MESSAGE :
                      journeyUserReducer.importStatus === 'LIMIT_EXCEEDED' ? userModuleConstants.LIMIT_EXCEEDED_MESSAGE : journeyUserReducer.importStatus === 'INVALID_IMP_TEMPLATE' ? <span> Please refer <a href={userModuleConstants.importUserInstLink} target="_blank">Import Users Instructions</a> for the latest template. </span> :
                        journeyUserReducer.importStatus === 'SUCCESS' ? <span>{journeyUserReducer.totalRecordsInserted} users were added! </span> : ''}
                  {journeyUserReducer.importStatus === 'ERROR' ?
                    <span>Download&nbsp;
                      <NavLink to="#">
                        <ExportErrorList
                          button={<span>Error Report</span>}
                          columnNames={userModuleConstants.columnNames}
                          rowData={journeyUserReducer.importUserErrorList}
                          filename="import-user-list"
                          excelSheetName="ImportUsers"
                        />
                      </NavLink> and try again.
                    </span> :
                    journeyUserReducer.importStatus === 'READY_TO_IMPORT' ? <span className="import-user-actions"><Button text="Import" variant="emphasis" isCompact onClick={this.handleOnClickImport} /> <span className="import-user-cancel" onClick={this.handleOnClickCancelImport} >Cancel</span> </span> : ''}
                </Alert> </div> : ''}
              {journeyUserReducer.showProgressIndicator ?
                <div className="import-user-progress-indicator"> <IconSpinner /> {journeyUserReducer.showImportProgressMessage ? 'Importing users. Please be patient.' : 'Checking your users list, please be patient'}</div> : ''
              }
              {isUserExportInProgress &&
                <Alert type="info" className="add-user-alert">{userModuleConstants.USER_EXPORT_IN_PROGRESS_MSG}</Alert>
              }
              {
                userExportError &&
                <Alert type="error" className="add-user-alert" onDismiss={() => this.dismissUserExportError(false)}>{userModuleConstants.USER_EXPORT_ERROR_MSG}</Alert>
              }
            </Grid.Column>
          </Grid.Row>
          <Grid.Row>
            <Grid.Column tiny={12} small={12} medium={12} large={12} huge={12}>
              {persistedDetails.privileges.includes('SEARCH_VIEW_USER') &&
                <div>
                  {journeyUserReducer.isJourneyUserListLoading ? <div style={{ marginTop: '2rem' }}><Loading padding="13.5" /> </div> :
                    <UserList journeyUserList={journeyUserReducer.filteredData} searchStringLength={this.state.searchStringLength} searchUser={this.searchUser}/>}
                </div>
              }
            </Grid.Column>
          </Grid.Row>
        </Grid>
      </React.Fragment>
    );
  }
}
