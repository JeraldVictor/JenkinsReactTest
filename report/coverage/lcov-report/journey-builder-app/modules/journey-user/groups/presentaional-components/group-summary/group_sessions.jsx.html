import React from 'react';
import { NavLink } from 'react-router-dom';
import { useSelector } from 'react-redux';
import Card from 'terra-card/lib/Card';
import IconEdit from 'terra-icon/lib/icon/IconEdit';
import Button from 'terra-button/lib/Button';
import ToggleSessionDetails from './toggle_session_details';
import NoData from '../../../../../utils/no_data';
import { NO_SESSION_ASSIGNMENT_DATA } from '../../../../../utils/group_summary_constants';
 
 
export default function GroupSessions(props) {
  const { groupSessions, isJourneyAssignmentComplete, isSessionAssignmentComplete } = useSelector(state => state.journeyGroupReducer);
  const { privileges } = useSelector(state => state.persistedDetails);
  return (
    <>
      <div className="edit_group_summary">
        {(isJourneyAssignmentComplete && isSessionAssignmentComplete && groupSessions.is_event_available && !privileges.includes('VIEW_FOR_SUPPORT_ADMIN')) ? <NavLink to={`${props.url}/session_assignment_details`}><Button icon={<IconEdit />} text="Edit Session Assignments" className="edit_session_assignments" /></NavLink> : <Button icon={<IconEdit />} text="Edit Session Assignments" isDisabled className="edit_session_assignments" />}
      </div>
      {groupSessions.group_session_details && groupSessions.group_session_details.data.length ? groupSessions.group_session_details.data.map(session => <ToggleSessionDetails key={session.id} sessionDetails={session} />) : <Card id="sessionAssignmentContainers" className="no-data-card no-data-properties no-data-pg5">
        <Card.Body isContentCentered className="align-empty-assignment no_group_session">
          <NoData no_data_text={groupSessions.is_event_available ? NO_SESSION_ASSIGNMENT_DATA.no_data_text : NO_SESSION_ASSIGNMENT_DATA.no_qualified_session} try_again_text={groupSessions.is_event_available && NO_SESSION_ASSIGNMENT_DATA.try_again_text} page_size={5} />
        </Card.Body>
      </Card>}
    </>
  );
}
