              Cancel

        return TECHNICAL_ERROR_MESSAGE

        return GROUP_LIMIT_EXCEEDED_MESSAGE

        return `${userCount} members were added!`
      case "ERROR":
        return (
          <span>
            Download&nbsp;
            <NavLink to="#">
              <ExportErrorList
                button={<span>Error Report</span>}
                columnNames={columnNames}
                rowData={journeyUserReducer.importUserErrorList}
                filename="import-user-list"
                excelSheetName="ImportUsers"
              />
            </NavLink>{" "}
            and try again.
          </span>
        )
      case "ERROR":
        return (
          <span className="import-user-actions">
            <Button
              text="Import"
              variant="emphasis"
              isCompact={true}
              onClick={handleOnClickImport}
            />
            <span
              className="import-user-cancel"
              onClick={handleOnClickCancelImport}
            >
              Cancel
            </span>
          </span>
        )
      default:
        return TECHNICAL_ERROR_MESSAGE;
    }
  }
 
  const fetchData = (state, instance) => {
    let data = {
      group_detail_id: journeyGroupReducer.groupId,
      search_string: '',
      page_number: state.page,
      page_size: state.pageSize,
      filters: {

        username: state.filtered.filter(key => key.id === 'username')[0]?.value ?? '',

        status: state.filtered.filter(key => key.id === 'status')[0]?.value ?? '',

    setSelectedAllStatus(selectedUsers);

  return (

          <Alert id="grpSummaryAlert" type={journeyGroupReducer.groupSummaryAlert.type} title={journeyGroupReducer.groupSummaryAlert.title}
            className="section_alert" onDismiss={() => dispatch(groupActions.dismissGrpSummaryError())}> {journeyGroupReducer.groupSummaryAlert.msg}
          </Alert>
        }
        {/* Error Alert */}
        {common.showSaveActivitiesError && <Alert id="serviceErrorAlert" className="section_alert" onDismiss={handleDismiss} type="error">{PASS_ACTIVITIES_ERROR_MESSAGE}</Alert>}
        {/* Success Alert upon Successful Save */}
        {common.showSaveActivitiesSuccess && !common.downloadReportData ? <IconSpinner /> : common.showSaveActivitiesSuccess && common.downloadReportData ? <Alert id="activitiesSavedSuccessAlert" className="section_alert" type="success" title="Success!"
          onDismiss={handleDismiss}> <span>{PASS_ACTIVITIES_SUCCESS_MESSAGE} Click
            <NavLink to="#">
              <ExportErrorList button={<span> Here</span>}
                columnNames={passActivityColumnNames}
                rowData={common.downloadReportData}
                filename={`Passed activities report for ${journeyGroupReducer.accessedGroupName} on ${moment(common.reportCreatedDate)
                  .tz(persistedDetails.timeZoneRegion)
                  .format('MMM. DD, YYYY HH:mm z')}`}
                excelSheetName="PassActivityReport"
              />
            </NavLink>  to view the report. </span></Alert> : ''}
        {common.activityFailedGroup && !common.downloadReportData ? <IconSpinner /> : common.activityFailedGroup && common.downloadReportData ? <Alert id="activitiesSavedfailedAlert" className="section_alert" type="error" title="Fail!"
          onDismiss={handleDismiss}> <span>{FAIL_ACTIVITIES_FAILURE_MESSAGE} Click
            <NavLink to="#">
              <ExportErrorList button={<span> Here</span>}
                columnNames={passActivityColumnNames}
                rowData={common.downloadReportData}
                filename={`Failed activities report for ${journeyGroupReducer.accessedGroupName} on ${moment(common.reportCreatedDate)
                  .tz(persistedDetails.timeZoneRegion)
                  .format('MMM. DD, YYYY HH:mm z')}`}
                excelSheetName="PassActivityReport"
              />
            </NavLink>  to view the report. </span></Alert> : ''}
        {/* Technical Error Alert  Ex: Network Issues (Auto dismissed after 10 secs)*/}
        {common.saveActivitiesTechnicalError && <Alert id="serviceErrorAlert" className="section_alert" type="error">{TECHNICAL_ERROR_MESSAGE}</Alert>}
        {/* Warning Alert for Activities in Progress*/}
        {common.isPassActivitiesInProgress && !common.otherAdminAccessed && <Alert type="info" className="section_alert" id="activitiesInProgressAlert">{PASS_ACTIVITIES_IN_PROGRESS_MESSAGE}</Alert>}
        {/* Warning Alert for Simultanous Click*/}
        {common.showSaveActivitiesInProgress && <Alert type="info" className="section_alert" id="simultanousSaveAlert">{PASS_ACTIVITIES_SIMULATANOUS_CLICK_MESSAGE}</Alert>}
        {((journeyGroupReducer.isJourneyAssignmentInProgress) || (!journeyGroupReducer.isJourneyAssignmentComplete &&
          !journeyGroupReducer.isJourneyAssignmentStatusLoading && journeyGroupReducer.isSessionAssignmentComplete)) &&
          <Alert type="info" className="section_alert" title=" ">{JOURNEY_ASSIGNMENT_IN_PROGRESS_TEXT}</Alert>}
        {journeyGroupReducer.isErrorOccurred &&
          <Alert type="error" className="section_alert">
            Service not Available
          </Alert>
        }
        {journeyGroupReducer.isMemberListLoading && <div className="import-user-progress-indicator">{" "}<IconSpinner /> Loading Details</div>}
        {journeyUserReducer.showProgressIndicator && (
          <div className="import-user-progress-indicator">
            {" "}
            <IconSpinner />{" "}
            {journeyUserReducer.showImportProgressMessage
              ? "Importing members. Please be patient"
              : "Checking your member list. Please be patient"}
          </div>
        )}
      </Grid.Column>
      <Grid.Column tiny={12} id="automationLoadMembersInstructionText" className="load_members_instruction_text">
        <a href="https://wiki.cerner.com/x/PZPOdw" target="_blank"> {LOAD_MEMBERS_INSTRUCTIONS} </a>
      </Grid.Column>
      <Grid.Column tiny={9} className="journey-user-action-buttons">
        {persistedDetails.privileges.includes('LOAD_GROUP_MEMBER') && <ImportMembersButton groupUserListLength={groupUserListLength} getAssignedJourneyStartDates={getAssignedJourneyStartDates} isDisabled={persistedDetails.privileges.includes('VIEW_FOR_SUPPORT_ADMIN')} />}
        {persistedDetails.privileges.includes('ADD_GROUP_MEMBER') && <AddMembersButton edit={props.edit} disableAdd={false} updateJourneyVersionInGroup={updateJourneyVersionInGroup} setSearchKeyWord={setSearchKeyWord} setDataList={setDataList} />}
        <PassActivitiesButton selectedUsers={selectedUsers} />
      </Grid.Column>
      <Grid.Column tiny={12}>
        {journeyGroupReducer.isMemberListLoading ? (
          <Loading padding="13.5" />
        ) : (
          <ReactTable
            data={journeyGroupReducer.paginatedList?.data}
            className="no-data-properties smaller-resize"
            noDataText={<NoGroupDataFound />}
            filterable={false}
            ref={reactTable}
            defaultSorted={[
              {
                id: 'name',
                desc: false,
              },
            ]}
            columns={columns}
            defaultPageSize={10}
            showPageSizeOptions
            onSortedChange={onSortViewChange}
            PaginationComponent={ServerSidePaginationComponent}
            rowMin={journeyGroupReducer.paginatedList?.start_index}
            rowMax={journeyGroupReducer.paginatedList?.end_index}
            rowTotal={journeyGroupReducer.paginatedList?.total_records}
            manual
            pages={journeyGroupReducer.paginatedList?.total_pages} //Displays the total no of pages
            loading={journeyGroupReducer.isGroupMembersLoading} //Displays the loading overlay when needed
              loadingText={'Fetching Data, Please Wait...'}
            onFetchData={fetchData}
            onPageChange={() => setSelectedAllStatus(selectedUsers)}
            onPageSizeChange={() => setSelectedAllStatus(selectedUsers)}
            onFilteredChange={() => setSelectedAllStatus(selectedUsers)}
            defaultFiltered={[{ // the current filters model
              id: 'status',
              value: 'ALL'
            }]}
          />)}
      </Grid.Column>
      <Grid.Column tiny={12} className="add-members-action-buttons">
        {!props.edit ? (<><Button
          text={props.edit ? 'Save' : 'Next'}
          className="next-button-with-margin"
          isDisabled={persistedDetails.privileges.includes('VIEW_FOR_SUPPORT_ADMIN') ||
            journeyGroupReducer.isJourneyAssigned
            ? journeyGroupReducer.loadingJourneyDetails
            : false
          }
          variant="emphasis"
          onClick={() => dispatch(groupActions.setStepperPosition(2))}
        />
          <Link to="/journey_user/groups">
            <Button
              text="Cancel"
              className="next-button-with-margin cancel-button"
            />
          </Link>
          <Button
            text="Previous"
            className="previous-button-with-margin"
            onClick={() => dispatch(groupActions.setStepperPosition(0))}
          /> </>) : (<Button
            text="Back"
            className="previous-button-with-margin"
            variant="emphasis"
            onClick={routeChange}
          />)}
      </Grid.Column>
    </Grid.Row>
  )
}
