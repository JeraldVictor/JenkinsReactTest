export const getMaxLearnerCount = (users) => {
  const stageMax = users.map(stageValues => Math.max(...stageValues.map(o => o)));
  return Math.max.apply(null, stageMax);
};
 
 
export const JourneyProgressView = ({ graphType, content, org, selectedJourney, selectedGroup, fromDate, toDate }) => {
  const count = graphType ? 100 : getMaxLearnerCount(content.users);
  const textToDisplay = graphType ? 'Percentage of Learners' : 'Number of Learners';
 
 
  const barRoot = {
 
    bindTo: 'barRoot',
    axis: {
      x: {
        show: true,
        label: 'Stage',
        lowerLimit: 0,
        upperLimit: 6,
        ticks: {
          values: [1, 2, 3, 4, 5],
          format: '.0f',
        },
      },
      y: {
        show: true,
        label: textToDisplay,
        lowerLimit: 0,
        upperLimit: count,
      },
    },
    showLabel: true,
    showLegend: true,
    showShapes: true,
    showVGrid: false,
    showHGrid: true,
  };
 
  graphType ? useCarbonBar(barRoot, contentBar(content)) : useCarbonBarCount(barRoot, contentBarCount(content));
 
 
  return (
    <React.Fragment>
      <div id={barRoot.bindTo} className="report-bar-container" />
      <div className="report-modals">
        <ModalManager>
          <StageProgressReportType org={org} selectedJourney={selectedJourney} selectedGroup={selectedGroup} fromDate={fromDate} toDate={toDate}/>
        </ModalManager>
      </div>
    </React.Fragment>);
};
