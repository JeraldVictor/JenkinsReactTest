import * as journeyActionTypes from '../../modules/journey-builder/journeys/actions/journey_action_types';
 
const initialState = {
  isJourneyListLoading: true,
  isErrorOccurred: false,
  journeyList: [],
  journeyListFromMaster: null,
  duplicateJourneyNames: [],
  loadSuccessful: null,
  status: null,
  searchText: '',
  isJourneyAdded: false,
  journeyVersionsList: null,
  linkId: '',
  accessCode: '',
  expirationDate: null,
  sharedJourneys: null,
  isLoadJourneyInProgress: false,
  selectedJourneysToLoad: {},
  duplicateErrorList: [],
  previewJourneyList: [],
  isPreviewJourneyListLoading: null,
  sharePreviewId: '',
  emailRecipients: '',
  emailMessage: null,
  activationStatusCancelled: false,
  hasRemovedDuplicateJourneys: false,
  showJourneyActivationStatusError: false,
  isContentReportDownloadInProgress: false,
  hasContentReportDownloadFailed: false,
  stageActivitySearchStatus: false,
  duplicateActivities: [],
  modifiedJourneyActivities: {},
  previousDuplicateJourneys: [],
  showFilterDataLoader: false,
  journeySearchFilterDetails: {},
  showJourneyListError: false,
};
 
const journeysReducer = (state = initialState, action) => {
  const newState = Object.assign({}, state);
 
  switch (action.type) {
    case journeyActionTypes.REQUEST_JOURNEY_LIST_SUCCEEDED:
      newState.journeyList = action.journeyList;
      newState.isJourneyListLoading = false;
      newState.showFilterDataLoader = false;
      return newState;
    case journeyActionTypes.REQUEST_JOURNEY_LIST_SUCCEEDED_FROM_MASTER_ORG:
      newState.journeyListFromMaster = action.journeyList;
      return newState;
    case journeyActionTypes.LOAD_JOURNEY_LIST_SUCCEEDED:
      newState.isLoadJourneyInProgress = false;
      newState.loadSuccessful = action.journeyList.loadSuccessful;
      newState.status = action.journeyList.status;
      newState.duplicateActivities = action.journeyList.duplicate_activities;
      newState.modifiedJourneyActivities  =
          (action.journeyList.duplicate_activities && action.journeyList.duplicate_activities.length !== 0 ?
              newState.modifiedJourneyActivities : {});
      newState.duplicateJourneyNames = action.journeyList.duplicate_journeys;
      newState.previousDuplicateJourneys =
          action.journeyList.duplicate_journeys && action.journeyList.duplicate_journeys.length !== 0 ?
              action.journeyList.duplicate_journeys : newState.previousDuplicateJourneys;
      newState.isJourneyAdded = true;
      action.journeyList.duplicate_journeys ? newState.isJourneyAdded = false ? newState.duplicateErrorList =
         (action.journeyList.duplicate_journeys).map(journey => Object.keys(journey)[0]) : null: true;
      return newState;
    case journeyActionTypes.RESET_LOAD_JOURNEY_SUCCESS:
      newState.loadSuccessful = false;
    case journeyActionTypes.REQUEST_JOURNEY_LIST:
      newState.searchText = action.searchText;
      return newState;
    case journeyActionTypes.REQUEST_JOURNEY_LIST_FROM_ORG:
      newState.searchText = action.searchText;
      return newState;
    case journeyActionTypes.RELOAD_JOURNEY_LIST:
      newState.isJourneyAdded = false;
      return newState;
    case journeyActionTypes.GET_JOURNEY_VERSIONS_LIST_SUCCEEDED:
      newState.journeyVersionsList = action.journeyVersionList;
      return newState;
    case journeyActionTypes.RESET_JOURNEY_VERSIONS_LIST:
      newState.journeyVersionsList = [];
      return newState;
    case journeyActionTypes.SAVE_SHARE_PREVIEW:
      newState.linkId = action.linkId;
      newState.accessCode = action.accessCode;
      newState.expirationDate = action.expirationDate;
      newState.sharedJourneys = action.sharedJourneys;
      return newState;
    case journeyActionTypes.SAVE_SHARE_PREVIEW_SUCCEEDED:
      newState.sharePreviewId = action.sharePreviewId;
      return newState;
    case journeyActionTypes.SEND_SHARE_PREVIEW_EMAILS:
      newState.emailRecipients = action.emailRecipients;
      newState.emailMessage = action.emailMessage;
      return newState;
    case journeyActionTypes.REQUEST_PREVIEW_JOURNEY_LIST:
      newState.isPreviewJourneyListLoading = true;
      return newState;
    case journeyActionTypes.REQUEST_PREVIEW_JOURNEY_LIST_SUCCEEDED:
      newState.previewJourneyList = action.previewJourneyList;
      newState.isPreviewJourneyListLoading = false;
      return newState;
    case journeyActionTypes.UPDATE_SHARE_PREVIEW:
      newState.sharePreviewId = action.sharePreviewId;
      newState.expirationDate = action.expirationDate;
      return newState;
    case journeyActionTypes.SET_LOAD_JOURNEY_IN_PROGRESS:
      newState.isLoadJourneyInProgress = true;
      return newState;
    case journeyActionTypes.RESET_DUPLICATE_JOURNEY_LIST:
      newState.duplicateJourneyNames = [];
      return newState;
    case journeyActionTypes.RESET_DUPLICATE_ACTIVITY_LIST:
      newState.duplicateActivities = [];
      newState.modifiedJourneyActivities = {};
      newState.previousDuplicateJourneys = [];
      return newState;
    case journeyActionTypes.EXCLUDE_DUPLICATE_JOURNEY:
      newState.duplicateJourneyNames = newState.duplicateJourneyNames.filter(journeyObj => Object.keys(journeyObj)[0] !== action.journeyId);
      newState.duplicateActivities = newState.duplicateActivities.filter(activityObj => activityObj.journey_id !== action.journeyId);
      delete newState.selectedJourneysToLoad[action.journeyId];
      newState.duplicateErrorList = newState.duplicateErrorList.filter(journeyId => journeyId !== action.journeyId);
      return newState;
    case journeyActionTypes.SET_NO_DUPLICATE_JOURNEY_FLAG:
      newState.hasRemovedDuplicateJourneys = action.hasRemovedDuplicateJourneys;
      return newState;
    case journeyActionTypes.SET_SELECTED_JOURNEYS_TO_LOAD:
      newState.selectedJourneysToLoad = action.journeysToLoad;
      return newState;
    case journeyActionTypes.SET_MODIFIED_JOURNEY_ACTIVITIES:
      newState.modifiedJourneyActivities = action.modifiedJourneyActivities;
      return newState;
    case journeyActionTypes.ADD_JOURNEY_TO_ERROR_LIST: {
      const currentJourneysWithError = [...newState.duplicateErrorList];
      currentJourneysWithError.push(action.journeyId);
      newState.duplicateErrorList = currentJourneysWithError;
      return newState;
    }
    case journeyActionTypes.DELETE_JOURNEY_FROM_ERROR_LIST: {
      const currentErrorJourneys = [...newState.duplicateErrorList];
      currentErrorJourneys.splice(currentErrorJourneys.indexOf(action.journeyId), 1);
      newState.duplicateErrorList = currentErrorJourneys;
      return newState;
    }
    case journeyActionTypes.RESET_IMPORT_JOURNEY_STATUS:
      newState.status = null;
      return newState;
    case journeyActionTypes.CANCEL_JOURNEY_ACTIVATION_MODAL:
      newState.activationStatusCancelled = true;
      return newState;
    case journeyActionTypes.SHOW_ACTIVATION_STATUS_ERROR:
      newState.showJourneyActivationStatusError = true;
      return newState;
    case journeyActionTypes.ACTIVATE_JOURNEY:
      newState.journeyList = action.journeyList;
      return newState;
    case journeyActionTypes.INACTIVATE_JOURNEY:
      newState.journeyList = action.journeyList;
      return newState;
    case journeyActionTypes.DOWNLOAD_CONTENT_REPORT_SUCCEEDED:
      newState.isContentReportDownloadInProgress = false;
      newState.hasContentReportDownloadFailed = !action.isContentReportDownloadSuccessful;
      return newState;
    case journeyActionTypes.SET_CONTENT_REPORT_DOWNLOAD_IN_PROGRESS:
      newState.isContentReportDownloadInProgress = action.isContentReportDownloadInProgress;
      return newState;
    case journeyActionTypes.RESET_CONTENT_REPORT_DOWNLOAD_ERROR_ALERT:
      newState.hasContentReportDownloadFailed = false;
      return newState;
    case journeyActionTypes.SET_JOURNEY_ACTIVITIES_SEARCH:
      newState.stageActivitySearchStatus = true;
      return newState;
    case journeyActionTypes.UNSET_JOURNEY_ACTIVITIES_SEARCH:
      newState.stageActivitySearchStatus = false;
      return newState;
    case journeyActionTypes.SHOW_FILTER_TABLE_LOADER:
      newState.showFilterDataLoader = true;
      return newState;
    case journeyActionTypes.SHOW_JOURNEY_TABLE_LOADER:
      newState.isJourneyListLoading = true;
      return newState;
     case journeyActionTypes.RETAIN_JOURNEY_SEARCH_FILTER:
        newState.journeySearchFilterDetails = Object.assign({}, action.filter_details);
        return newState;
      case journeyActionTypes.SHOW_REQUEST_JOURNEY_LIST_ERROR_MESSAGE:
          newState.showJourneyListError = true;
        return newState;
      case journeyActionTypes.RESET_JOURNEY_TABLE_LOADER:
        newState.isJourneyListLoading = false;
        return newState;
    default:
      return state;
  }
};
 
export default journeysReducer;
